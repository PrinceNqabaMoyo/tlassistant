import os
import pandas as pd
import json
import re

# --- Configuration ---
# The folder where your CSV topic files are located.
# This should be the same as your RAG ingestion folder.
DOCS_DIR = "curriculum_docs"

# The path where the final JavaScript data file will be saved.
# Adjust this path to point to your React app's 'src' folder.
OUTPUT_PATH = "../caps-ai-frontend/src/curriculumData.js" 

def create_curriculum_data_file():
    """
    Scans a directory for topic CSV files, processes them, and generates
    a single JavaScript data file for the frontend.
    """
    print("Starting curriculum data build process...")
    
    # This will hold the final nested dictionary structure.
    curriculum_data = {}

    # Regex to match the topic CSV filenames (e.g., "Mathematics_Gr9_Topics.csv")
    topic_file_pattern = re.compile(r"(.+)_Gr(\d+)_Topics\.csv")

    try:
        # Ensure the output directory exists
        output_dir = os.path.dirname(OUTPUT_PATH)
        if not os.path.exists(output_dir):
            os.makedirs(output_dir)
            print(f"Created output directory: {output_dir}")

        for filename in os.listdir(DOCS_DIR):
            match = topic_file_pattern.match(filename)
            if match:
                subject_raw, grade = match.groups()
                # Clean up subject name for display
                subject = subject_raw.replace('MathematicalLiteracy', 'Mathematical Literacy').replace('TechnicalMathematics', 'Technical Mathematics')
                
                print(f"-> Processing file: {filename} for Subject: {subject}, Grade: {grade}")

                filepath = os.path.join(DOCS_DIR, filename)
                
                # UPDATED: Use a semicolon as the separator to match the file format.
                df = pd.read_csv(filepath, encoding='utf-8', sep=';')

                # Ensure the subject key exists in the main dictionary
                if subject not in curriculum_data:
                    curriculum_data[subject] = {}
                
                # Ensure the grade key exists for the subject
                if grade not in curriculum_data[subject]:
                    curriculum_data[subject][grade] = {}

                # Iterate over each column (which represents a main topic and its subtopics)
                for main_topic in df.columns:
                    # Get all subtopics in the column, drop any empty cells (NaN), and convert to a list
                    subtopics = df[main_topic].dropna().tolist()
                    curriculum_data[subject][grade][main_topic] = subtopics

        if not curriculum_data:
            print("\nWarning: No topic CSV files were found or processed. The output file will be empty.")
            return

        # Convert the Python dictionary to a JSON string for writing to the file
        # The `indent=4` makes the output file readable.
        json_string = json.dumps(curriculum_data, indent=4)

        # Write the data to the JavaScript file, exporting it as a constant
        with open(OUTPUT_PATH, 'w', encoding='utf-8') as f:
            f.write("// This file is auto-generated by the build_curriculum_data.py script.\n")
            f.write("// Do not edit this file manually.\n\n")
            f.write("export const curriculumData = ")
            f.write(json_string)
            f.write(";")

        print(f"\nSuccess! Curriculum data has been written to: {OUTPUT_PATH}")

    except FileNotFoundError:
        print(f"\nError: The directory '{DOCS_DIR}' was not found. Please make sure it exists and contains your CSV files.")
    except Exception as e:
        print(f"\nAn unexpected error occurred: {e}")

if __name__ == "__main__":
    # Before running, you may need to install pandas:
    # pip install pandas
    create_curriculum_data_file()
